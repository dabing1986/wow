function()
    
    --Rage buffer T7
    local spellRageBufferT7
    local spellCDT7
    if aura_env.spellCostT7 then
        local triggerStateT7 = WeakAuras.GetTriggerStateForTrigger(aura_env.id, aura_env.triggerIdT7)
        local expirationTimeT7 = triggerStateT7[""].expirationTime
        if expirationTimeT7 ~= 0 then
            spellCDT7 = expirationTimeT7 - GetTime()
        else
            spellCDT7 = 0
        end
        spellRageBufferT7 = min(aura_env.spellCostT7 / 6 * (6 - spellCDT7 + 1.5), aura_env.spellCostT7)
    else
        spellRageBufferT7 = 0
    end
    
    --CD Whirlwind
    local triggerStateWW = WeakAuras.GetTriggerStateForTrigger(aura_env.id, 7)
    local expirationTimeWW = triggerStateWW[""].expirationTime
    local spellCDWW
    if expirationTimeWW ~= 0 then
        spellCDWW = expirationTimeWW - GetTime()
    else
        spellCDWW = 0
    end
    
    return UnitPower("player") >= spellRageBufferT7 + aura_env.spellCostWW and
    (
        aura_env.triggerIdT7 == nil
        or spellCDWW <= spellCDT7 - 1.2
        or aura_env.triggerIdT7 == 6 and not IsEquippedItemType("Shields")
    ) or not aura_env.spellCostT7
    
end
